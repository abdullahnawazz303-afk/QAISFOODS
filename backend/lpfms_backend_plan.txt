Stored Procedures & FunctionsAll critical business operations run as atomic PostgreSQL stored procedures. This ensures data consistency â€” if any step fails, the entire transaction rolls back.6.1 Create Sale (Atomic)This is the most critical procedure. It creates a sale, deducts inventory, updates the customer ledger, and records cash flow â€” all in one atomic transaction.CREATE OR REPLACE FUNCTION create_sale(  p_customer_id   UUID,  p_sale_date     DATE,  p_items         JSONB,   -- [{batch_id, quantity_kg, sale_price_per_kg}]  p_amount_paid   NUMERIC,  p_notes         TEXT,  p_created_by    UUID,  p_cash_day_id   UUID     -- link to today's cash register)RETURNS UUID AS $$DECLARE  v_sale_id       UUID;  v_total         NUMERIC := 0;  v_item          JSONB;  v_subtotal      NUMERIC;  v_sale_ref      VARCHAR;  v_cust_balance  NUMERIC;BEGIN  -- Generate sale reference  v_sale_ref := 'SALE-' || TO_CHAR(NOW(), 'YYYYMMDD') || '-'                || LPAD(nextval('sale_seq')::TEXT, 3, '0');  -- Calculate total from items  FOR v_item IN SELECT * FROM jsonb_array_elements(p_items) LOOP    v_subtotal := (v_item->>'quantity_kg')::NUMERIC *                  (v_item->>'sale_price_per_kg')::NUMERIC;    v_total := v_total + v_subtotal;  END LOOP;  -- Insert sale header  INSERT INTO sales(sale_ref, customer_id, sale_date, total_amount,                    amount_paid, notes, created_by)  VALUES(v_sale_ref, p_customer_id, p_sale_date, v_total,         p_amount_paid, p_notes, p_created_by)  RETURNING id INTO v_sale_id;  -- Insert line items + deduct inventory  FOR v_item IN SELECT * FROM jsonb_array_elements(p_items) LOOP    -- Validate stock availability    IF (SELECT remaining_qty_kg FROM inventory_batches        WHERE id = (v_item->>'batch_id')::UUID)        < (v_item->>'quantity_kg')::NUMERIC THEN      RAISE EXCEPTION 'Insufficient stock for batch %',                       v_item->>'batch_id';    END IF;    INSERT INTO sale_items(sale_id, batch_id, quantity_kg, sale_price_per_kg)    VALUES(v_sale_id, (v_item->>'batch_id')::UUID,           (v_item->>'quantity_kg')::NUMERIC,           (v_item->>'sale_price_per_kg')::NUMERIC);    -- Deduct from inventory batch    UPDATE inventory_batches    SET remaining_qty_kg = remaining_qty_kg - (v_item->>'quantity_kg')::NUMERIC,        updated_at = NOW()    WHERE id = (v_item->>'batch_id')::UUID;    -- Log stock movement    INSERT INTO inventory_movements(batch_id, movement_type,                  quantity_kg, reference_type, reference_id, created_by)    VALUES((v_item->>'batch_id')::UUID, 'OUT',           (v_item->>'quantity_kg')::NUMERIC, 'SALE', v_sale_id, p_created_by);  END LOOP;  -- Get current customer balance  SELECT current_balance INTO v_cust_balance FROM customers  WHERE id = p_customer_id;  -- Add to customer ledger (debit = they owe us more)  INSERT INTO customer_ledger(customer_id, entry_date, transaction_type,    description, debit, credit, running_balance, reference_type,    reference_id, created_by)  VALUES(p_customer_id, p_sale_date, 'SALE',    'Sale: ' || v_sale_ref, v_total, 0,    v_cust_balance + v_total, 'SALE', v_sale_id, p_created_by);  -- Update customer balance  UPDATE customers SET current_balance = current_balance + v_total,    updated_at = NOW() WHERE id = p_customer_id;  -- If cash was paid, record in cash register  IF p_amount_paid > 0 AND p_cash_day_id IS NOT NULL THEN    INSERT INTO cash_entries(cash_day_id, entry_type, category,      amount, description, reference_type, reference_id, created_by)    VALUES(p_cash_day_id, 'IN', 'SALE_REVENUE', p_amount_paid,      'Payment for ' || v_sale_ref, 'SALE', v_sale_id, p_created_by);    UPDATE cash_days SET total_cash_in = total_cash_in + p_amount_paid    WHERE id = p_cash_day_id;  END IF;  RETURN v_sale_id;END;$$ LANGUAGE plpgsql;6.2 Close Business DayCREATE OR REPLACE FUNCTION close_business_day(  p_cash_day_id UUID,  p_closed_by   UUID)RETURNS NUMERIC AS $$  -- returns closing balanceDECLARE  v_opening   NUMERIC;  v_cash_in   NUMERIC;  v_cash_out  NUMERIC;  v_closing   NUMERIC;BEGIN  -- Validate day is open  IF NOT EXISTS (SELECT 1 FROM cash_days                 WHERE id = p_cash_day_id AND status = 'OPEN') THEN    RAISE EXCEPTION 'Day is already closed or does not exist';  END IF;  SELECT opening_balance, total_cash_in, total_cash_out  INTO v_opening, v_cash_in, v_cash_out  FROM cash_days WHERE id = p_cash_day_id;  v_closing := v_opening + v_cash_in - v_cash_out;  -- Lock the day  UPDATE cash_days  SET status = 'CLOSED', closing_balance = v_closing,      closed_at = NOW(), closed_by = p_closed_by  WHERE id = p_cash_day_id;  -- Lock all entries for this day  UPDATE cash_entries SET is_locked = TRUE  WHERE cash_day_id = p_cash_day_id;  RETURN v_closing;END;$$ LANGUAGE plpgsql;6.3 Mark Advance Booking as DeliveredCREATE OR REPLACE FUNCTION deliver_advance_booking(  p_booking_id UUID,  p_created_by UUID)RETURNS VOID AS $$DECLARE  v_item        RECORD;  v_batch_ref   VARCHAR;  v_vendor_id   UUID;BEGIN  SELECT vendor_id INTO v_vendor_id  FROM advance_bookings WHERE id = p_booking_id;  -- Create inventory batch for each booking item  FOR v_item IN    SELECT * FROM booking_items WHERE booking_id = p_booking_id  LOOP    v_batch_ref := 'BATCH-' || TO_CHAR(NOW(),'YYYYMMDD')                   || '-' || LPAD(nextval('batch_seq')::TEXT,3,'0');    INSERT INTO inventory_batches(      batch_ref, item_name, grade, vendor_id, purchase_price_per_kg,      quantity_kg, remaining_qty_kg, purchase_date,      advance_booking_id, created_by)    VALUES(      v_batch_ref, v_item.item_name, v_item.grade, v_vendor_id,      v_item.agreed_price_per_kg, v_item.quantity_kg,      v_item.quantity_kg, CURRENT_DATE,      p_booking_id, p_created_by);    INSERT INTO inventory_movements(      batch_id, movement_type, quantity_kg,      reference_type, reference_id, created_by)    VALUES(lastval()::UUID, 'IN', v_item.quantity_kg,           'BOOKING_DELIVERY', p_booking_id, p_created_by);  END LOOP;  -- Update booking status  UPDATE advance_bookings  SET status = 'DELIVERED', delivered_at = NOW(), updated_at = NOW()  WHERE id = p_booking_id;END;$$ LANGUAGE plpgsql;6.4 Handle Cheque BounceCREATE OR REPLACE FUNCTION bounce_cheque(  p_cheque_id    UUID,  p_bounce_reason TEXT,  p_created_by   UUID)RETURNS VOID AS $$DECLARE  v_cheque RECORD;  v_balance NUMERIC;BEGIN  SELECT * INTO v_cheque FROM cheques WHERE id = p_cheque_id;  IF v_cheque.status != 'PENDING' THEN    RAISE EXCEPTION 'Only PENDING cheques can be bounced';  END IF;  -- Mark cheque as bounced  UPDATE cheques  SET status = 'BOUNCED', bounced_at = NOW(),      bounce_reason = p_bounce_reason, updated_at = NOW()  WHERE id = p_cheque_id;  -- Get current vendor balance  SELECT current_balance INTO v_balance  FROM vendors WHERE id = v_cheque.vendor_id;  -- Re-add to vendor ledger (debt is back)  INSERT INTO vendor_ledger(vendor_id, entry_date, transaction_type,    description, debit, credit, running_balance,    reference_type, reference_id, created_by)  VALUES(v_cheque.vendor_id, CURRENT_DATE, 'CHEQUE_BOUNCED',    'Cheque #' || v_cheque.cheque_number || ' bounced: ' || p_bounce_reason,    0, v_cheque.amount, v_balance + v_cheque.amount,    'CHEQUE', p_cheque_id, p_created_by);  -- Update vendor balance  UPDATE vendors  SET current_balance = current_balance + v_cheque.amount,      updated_at = NOW()  WHERE id = v_cheque.vendor_id;END;$$ LANGUAGE plpgsql;7.
